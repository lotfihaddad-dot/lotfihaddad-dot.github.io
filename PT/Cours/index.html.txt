    </div>

    <script>
        // Détecter automatiquement le nom du dossier courant (MP, MPSI, etc.)
        const chemin = window.location.pathname;
        const nomDossier = chemin.split('/').filter(p => p)[0] || 'documents';
        document.getElementById('titre-filiere').innerHTML = `📚 Documents ${nomDossier}`;
        
        // Extensions de fichiers à afficher
        const extensions = ['pdf', 'doc', 'docx', 'txt', 'odt', 'jpg', 'png', 'gif', 'xls', 'xlsx'];
        
        // Récupérer la liste des fichiers dans le dossier courant
        const url = window.location.pathname.substring(0, window.location.pathname.lastIndexOf('/')) + '/';
        
        fetch(url)
            .then(response => response.text())
            .then(html => {
                const parser = new DOMParser();
                const doc = parser.parseFromString(html, 'text/html');
                const links = doc.querySelectorAll('a');
                const ul = document.getElementById('liste-docs');
                let fichiersTrouves = 0;
                
                links.forEach(link => {
                    let href = link.getAttribute('href');
                    if (href && !href.startsWith('?') && href !== './' && href !== '../' && href !== '/' && href !== 'index.html' && !href.includes('index.html')) {
                        // Vérifier l'extension
                        const ext = href.split('.').pop().toLowerCase();
                        if (extensions.includes(ext)) {
                            fichiersTrouves++;
                            const li = document.createElement('li');
                            const a = document.createElement('a');
                            // Décoder le nom du fichier (pour les accents)
                            let nomFichier = decodeURIComponent(href);
                            // Nettoyer le nom
                            nomFichier = nomFichier.replace(/_/g, ' ').replace(/\.pdf$/i, '');
                            a.href = href;
                            a.textContent = nomFichier;
                            li.appendChild(a);
                            ul.appendChild(li);
                        }
                    }
                });
                
                if (fichiersTrouves === 0) {
                    ul.innerHTML = '<li style="text-align:center; color:#7f8c8d;">📭 Aucun document pour le moment. Les fichiers seront bientôt ajoutés.</li>';
                } else {
                    // Ajouter un compteur
                    const compteur = document.createElement('p');
                    compteur.style.textAlign = 'center';
                    compteur.style.color = '#7f8c8d';
                    compteur.style.marginTop = '20px';
                    compteur.innerHTML = `📄 ${fichiersTrouves} document(s) disponible(s)`;
                    ul.parentNode.insertBefore(compteur, ul.nextSibling);
                }
            })
            .catch(() => {
                document.getElementById('liste-docs').innerHTML = '<li>⚠️ Impossible de charger la liste des fichiers.</li>';
            });
    </script>
